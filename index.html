CSS Grid Layout
The best practice
How to start using Grid?

set property on grid container


						.container {
							display: grid;
						}

set columns value


						.container {
							display: grid;
							grid-template-columns: 200px 1fr;
						}

use percentages


						.container {
							display: grid;
							grid-template-columns: 33.33% 33.33% 33.33%;
						}

or fr


						.container {
							display: grid;
							grid-template-columns: 1fr 1fr 1fr;
						}

set rows value


						.container {
							display: grid;
							grid-template-columns: 200px 1fr 200px;
							grid-template-rows: 100px 150px 50px 100px;
						}

minmax function


					  .container {
							display: grid;
							grid-template-columns: 300px minmax(100px, 1fr) 10fr;
							grid-template-rows: 100px 150px 50px 100px;
							gap: 10px;
						}
Various properties

general properties


							.container {
								display: grid;
								grid-template-columns: 1fr 1fr;
								grid-auto-rows: minmax(100px, 1fr);
								gap: 10px;
								align-items: end;
								justify-items: end;
							}

personal properties


							.item1 {
								justify-self: start;
								align-self: center;
							}
The position of the element on the page

							.container {
								display: grid;
								grid-template-columns: 1fr 1fr;
								grid-auto-rows: minmax(100px 1fr);
								gap: 10px;
							}
					
							.item3 {
								grid-column-start: 2;
								grid-column-end: 3;
								grid-row-start: 1;
								grid-row-end: 4;
							}
Short entry

              .item3 {
                grid-column: 2/3;
                grid-row: 3/6;
              }
z-index

              .item3 {
                grid-column-start: 2;
                grid-column-end: 3;
                grid-row-start: 1;
                grid-row-end: 4;
                background-color: greenyellow;
                z-index: 2;
              }

              .item4 {
                background-color: red;
                grid-column-start: 1;
                grid-column-end: 3;
                grid-row-start: 2;
                grid-row-end: 3;
                z-index: 3;
              }
Grid template areas

              .container {
                display: grid;
                grid-template-areas: 'header header'
                                     'sidebar main'
                                     'footer footer';
              }
						

              .header {
                grid-area: header;
              }

              .sidebar {
                grid-area: sidebar;
              }

              .main {
                grid-area: main;
              }

              .footer {
                grid-area: footer;
              }
